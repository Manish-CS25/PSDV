print(f"Sum of even numbers: {even_sum}")

# 3. Compute the product of all odd numbers in the list.
odd_product = 1
for num in natural_numbers:
    if num % 2 != 0:
        odd_product *= num
print(f"Product of odd numbers: {odd_product}")

# 4. Calculate the mean, median, and standard deviation of the list.

mean = sum(natural_numbers) / len(natural_numbers)
sorted_numbers = sorted(natural_numbers)
middle = len(sorted_numbers) // 2
median = (sorted_numbers[middle] + sorted_numbers[~middle]) / 2
std_dev = (sum((x - mean) ** 2 for x in natural_numbers) / len(natural_numbers)) ** 0.5

print(f"Mean: {mean}, Median: {median}, Standard Deviation: {std_dev}")

# Q3.Fibonacci Sequence:

# 1. Write a Python function that takes a number n and returns a list of the first n Fibonacci numbers
def fibonacci(n):
    fibno_list = [0, 1]
    for i in range(2, n):
        fibno_list.append(fibno_list[i - 1] + fibno_list[i - 2])
    return fibno_list

# 2.Plot the Fibonacci sequence using a line graph.

# Q 10. Character Frequency:
# Write a function that accepts a string and:
def char_freq(string):
  # Converts all characters to lowercase.
  string.lower()
  freq = {}
  # Counts the frequency of each character.
  for char in string:
    if char in freq:
      freq[char] += 1
    else:
      freq[char] = 1
  # Returns a dictionary with characters as keys and their counts as values.
  print(f"Character frequency: {freq}")
  return freq


# Visualize the character frequencies using a bar chart

# Write a Python function to find all anagrams of a word in a given list.
# Input: word = "listen", words = ["enlist", "google", "inlets", "banana"]
# Output: ["enlist", "inlets"]

word = "listen"
words = ["enlist", "google", "inlets", "banana"]

def find_anagrams(word, words):
  anagrams = []
  for w in words:
    if sorted(word) == sorted(w):
      anagrams.append(w)
  return anagrams

print(f"Anagrams: {find_anagrams(word, words)}")

# Q22. Normalization:
# Write a Python function to normalize a list of numbers to a range of 0 to 1.
myList = [13 , 14 , 10 , 3 , 5, 9 , 11]
def normalize(myList):
  min = myList[0]
  max = myList[0]
  for num in myList:
    if num < min:
      min = num
    if num > max:
      max = num
  for i in range(len(myList)):
    myList[i] = (myList[i] - min) / (max - min)
  return myList
print(f"Normalized list: {normalize(myList)}")
# Visualize the normalized data using a line plot.
import matplotlib.pyplot as plt
fig_width = 9
fig_height = 6
plt.figure(figsize=(fig_width, fig_height))
plt.plot(normalize(myList))
plt.show()
